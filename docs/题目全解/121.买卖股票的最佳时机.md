```typescript
function maxProfit(prices: number[]): number {
    let pre:number = prices[0];
    let cur:number = prices[0];
    let low:number = prices[0];
    let res:number = 0;

    for(let i =0;i<prices.length;i++){
        cur = prices[i];
        if(cur>pre){
            low = Math.min(low,pre)
            res = Math.max(res,cur-low,cur-pre)
        }
        pre = cur;
    }

    return res;
};
```

```typescript
function maxProfit(prices: number[]): number {
  let low:number = Number.POSITIVE_INFINITY;
  let res = 0

  for (let i = 0; i < prices.length; i++) {
    if (prices[i] < low) {
      low = prices[i];
    }

    res = Math.max(res, prices[i] - low);
  }

  return res;
}
```

```typescript
function maxProfit(prices: number[]): number {
  let dp:number[][] = new Array(prices.length)
  for(let i=0;i<prices.length;i++){
      dp[i] = new Array(2).fill(0)
  }
  dp[0][0] = -prices[0]
  dp[0][1] = 0

  for(let i=1;i<prices.length;i++){
      dp[i][0] = Math.max(dp[i-1][0],-prices[i])
      dp[i][1] = Math.max(dp[i-1][1],dp[i-1][0]+prices[i])
  }
  return dp[prices.length-1][1]
}
```
