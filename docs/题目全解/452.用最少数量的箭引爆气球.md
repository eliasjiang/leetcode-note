function findMinArrowShots(points: number[][]): number {
    points.sort((a,b) => {
        if (a[0]!==b[0]){
            return a[0] - b[0]
        }else{
            return a[1] - b[1]
        }
    })
    
    
    let stack:number[][] = []

    for (const point of points){

        if (stack.length == 0){
            stack.push(point)
        }else{
            let top_point = stack[stack.length - 1]
            if(top_point[1] >= point[0]){
                stack.pop()
                stack.push([Math.max(top_point[0],point[0]),Math.min(top_point[1],point[1])])
            }else{
                stack.push(point)
            }
        }
    }
    
    return stack.length

function findMinArrowShots(points: number[][]): number {
    points.sort((a,b) => {
        if (a[0]!==b[0]){
            return a[0] - b[0]
        }else{
            return a[1] - b[1]
        }
    })
    
    
    let arrow_num:number = 1;

    for (let i:number = 1; i < points.length; i++){
        if(points[i][0] > points[i-1][1]){
            arrow_num++;
        }else{
            points[i][1] = Math.min(points[i][1],points[i-1][1])
        }
    }
    
    return arrow_num
};
