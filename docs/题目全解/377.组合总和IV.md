function combinationSum4(nums: number[], target: number): number {
    nums.sort((a, b) => a - b);
    let dp:number[] = new Array(target + 1).fill(0);
    dp[0] = 1;
    for (let i = 0; i < target+1; i++) {
        for(let j = 0;j < nums.length;j++){
            if(nums[j] <= i){
                dp[i] += dp[i - nums[j]];
            }
        }
    }
    return dp[target];
};
