```typescript
function coinChange(coins: number[], amount: number): number {
  let dp: number[] = new Array(amount + 1).fill(0);
  for (let i = 0; i < amount + 1; i++) {
    if (i % coins[0] === 0) {
      dp[i] = i / coins[0];
    } else {
      dp[i] = -1;
    }
  }
  for (let i = 0; i < coins.length; i++) {
    for (let j = 0; j < amount + 1; j++) {
      if (j - coins[i] >= 0) {
        if (dp[j] === -1 && dp[j - coins[i]] != -1) {
          dp[j] = dp[j - coins[i]] + 1;
        }
        if (dp[j] !== -1 && dp[j - coins[i]] !== -1) {
          dp[j] = Math.min(dp[j], dp[j - coins[i]] + 1);
        }
      }
    }
  }
  return dp[amount];
}
```

```typescript
function coinChange(coins: number[], amount: number): number {
  let dp: number[] = new Array(amount + 1).fill(Infinity);
  dp[0] = 0;
  for (let i: number = 0; i < coins.length; i++) {
    for (let j: number = coins[i]; j <= amount; j++) {
      if (dp[j - coins[i]] !== Infinity) {
        dp[j] = Math.min(dp[j], dp[j - coins[i]] + 1);
      }
    }
  }
  if (dp[amount] === Infinity) {
    return -1;
  }
  return dp[amount];
}
```
