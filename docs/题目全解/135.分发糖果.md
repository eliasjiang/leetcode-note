
贪心法 两边分别比较
```typescript
function candy(ratings: number[]): number {
    let candy_left: number[] = new Array(ratings.length).fill(1);
    let candy_right: number[] = new Array(ratings.length).fill(1);
    let candy_num:number = 0;
    for (let i=0;i<ratings.length-1;i++){
        if(ratings[i+1]>ratings[i]){
            candy_left[i+1] = candy_left[i] + 1;
        }
    }

    for (let j=ratings.length-1;j>0;j--){
        if(ratings[j-1]>ratings[j]){
            candy_right[j-1] = candy_right[j] + 1;
        }
    }

    for (let k=0;k<ratings.length;k++){
        candy_num += Math.max(candy_left[k],candy_right[k])
    }

    return candy_num
};
```
