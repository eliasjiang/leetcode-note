function maxAreaOfIsland(grid: number[][]): number {
    const dir:number[][] = [[0,1],[1,0],[-1,0],[0,-1]];
    const [m,n]:[number,number] = [grid.length,grid[0].length];
    
    function dfs(grid:number[][],visited:boolean[][],x:number,y:number):number{
        if(visited[x][y] || grid[x][y] === 0){
            return 0;
        }
        visited[x][y] = true;

        let area:number = 1;
        for(let i=0;i<4;i++){
            let nextX:number = x + dir[i][0]
            let nextY:number = y + dir[i][1]

            if(nextX < 0 || nextX >= m || nextY <0 || nextY >= n){
                continue
            }

            area += dfs(grid,visited,nextX,nextY);
            
        }

        return area


    }

    const visited:boolean[][] = Array.from({length:m},_ => new Array(n).fill(false));

    let maxArea: number = 0;

    for(let i=0;i<m;i++){
        for(let j=0;j<n;j++){
            maxArea = Math.max(dfs(grid,visited,i,j),maxArea)
        }
    }

    return maxArea
 
};

function maxAreaOfIsland(grid: number[][]): number {
    const dir:number[][] = [[0,1],[1,0],[-1,0],[0,-1]];
    const [m,n]:[number,number] = [grid.length,grid[0].length];
    
    function bfs(grid:number[][],visited:boolean[][],x:number,y:number):number{
        const queue:[number,number][] = [];
        queue.push([x,y])
        visited[x][y] = true
        let area:number = 0;
        
        while(queue.length != 0){
            const [xout,yout] = queue.shift();
            
            area += 1
            for(let i=0;i<4;i++){
                let nextX:number = xout + dir[i][0]
                let nextY:number = yout + dir[i][1]

                if(nextX <0 || nextX >= m || nextY <0 || nextY >= n){
                    continue;
                }

                if(!visited[nextX][nextY] && grid[nextX][nextY] === 1){
                    visited[nextX][nextY] = true
                    queue.push([nextX,nextY])

                }
            }
        }

        return area


    }

    const visited:boolean[][] = Array.from({length:m},_ => new Array(n).fill(false));

    let maxArea: number = 0;

    for(let i=0;i<m;i++){
        for(let j=0;j<n;j++){
            if(!visited[i][j] && grid[i][j] === 1){
                maxArea = Math.max(bfs(grid,visited,i,j),maxArea)
            }
        }
    }

    return maxArea
 
};
