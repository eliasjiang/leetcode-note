function minCostClimbingStairs(cost: number[]): number {
    let len:number = cost.length;
    let dp:number[] = new Array(len+1)
    dp[0] = 0;
    dp[1] = 0;
    for(let i = 2; i <= len; i++) {
        dp[i] = Math.min(dp[i-1] + cost[i-1], dp[i-2] + cost[i-2])
    }
    return dp[dp.length-1]
};
