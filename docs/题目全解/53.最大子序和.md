
贪心算法

```python
class Solution:
    def maxSubArray(self, nums: List[int]) -> int:
        res = float('-inf')
        count = 0
        for i in range(0,len(nums)):
            count += nums[i]
            
            if count > res:
                res = count
            if count <= 0:
                count = 0
        return res
```

```typescript
function maxSubArray(nums: number[]): number {
    let res = nums[0];
    let sum = nums[0];
    for(let i = 1;i<nums.length;i++){
        if(sum < 0){
            sum = nums[i];
        }else{
            sum += nums[i];
        }
        res = Math.max(res,sum);
    }
    return res;
};
```

动态规划

```typescript
function maxSubArray(nums: number[]): number {
    let dp:number[] = new Array(nums.length)
    dp[0] = nums[0];
    let res:number = nums[0]

    for(let i=1;i<nums.length;i++){
        dp[i] = dp[i-1] > 0 ? dp[i-1] + nums[i] : nums[i]
        res = Math.max(dp[i],res)
    }

    return res
};
```
