function lastStoneWeightII(stones: number[]): number {
    stones.sort((a,b) => a-b)
    let left:number;
    let right:number;
    let temp:number[];
    while(stones.length > 1){
        left = 0;
        right = stones.length-1;
        temp = [];
        while(left <= right){
            if(left === right){
                temp.push(stones[left])
            }else{
                temp.push(stones[right] - stones[left])
            }
        }
        stones = temp;
    }

    return stones[0]
};

function lastStoneWeightII(stones: number[]): number {
    let sum:number = stones.reduce((a,b)=>a+b,0);
    let target:number = Math.floor(sum/2);
    let dp:number[][] = []
    for (let i = 0; i < stones.length; i++) {
        dp.push(new Array(target+1).fill(0));
    }
    for (let j = stones[0]; j < target+1; j++) {
        dp[0][j] = stones[0];
    }
    for (let i = 1; i < stones.length; i++) {
        for (let j = 1; j < target+1; j++) {
            if(j>=stones[i]){
                dp[i][j] = Math.max(dp[i-1][j],dp[i-1][j-stones[i]]+stones[i]);
            }else{
                dp[i][j] = dp[i-1][j];
            }
        }
    }

    return sum - dp[stones.length-1][target] - dp[stones.length-1][target];
};
