interface cashier{
    fiveDollars:number;
    tenDollars:number;
}
function lemonadeChange(bills: number[]): boolean {
    let cashier:cashier = {
        fiveDollars:0,
        tenDollars:0,
    }

    
    for(const bill of bills){
        
        switch(bill){
            case 5:
                cashier.fiveDollars += 1;
                break;
            case 10:
                if(cashier.fiveDollars >= 1){
                    cashier.fiveDollars -= 1;
                    cashier.tenDollars += 1;
                }else{
                    return false
                }
                break;
            case 20:
                if (cashier.tenDollars >= 1 && cashier.fiveDollars >=1){
                    cashier.tenDollars -= 1;
                    cashier.fiveDollars -= 1;
                }else if(cashier.fiveDollars >=3){
                    cashier.fiveDollars -= 3;
                }else{
                    return false;
                }
                break;
            default:
                break; 
        }

    }

    return true
};
