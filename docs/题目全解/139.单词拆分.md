function wordBreak(s: string, wordDict: string[]): boolean {
  let wordSet = new Set<string>(wordDict);

  let dp = new Array(s.length + 1).fill(false);
  dp[0] = true;

  for (let i = 1; i <= s.length; i++) {
    for (let j = 0; j < i; j++) {
      if (dp[j] && wordSet.has(s.substring(j, i))) {
        dp[i] = true;
      }
    }
  }

  return dp[s.length];
}
