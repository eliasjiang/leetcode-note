后序遍历

```typescript
function traversal(node: TreeNode | null): [number, number] {
    if(node == null) return [0, 0]
    const left:[number, number] = traversal(node.left)
    const right:[number, number] = traversal(node.right)
    const rob: number = node.val + left[0] + right[0]
    const notRob: number = Math.max(left[0], left[1]) + Math.max(right[0], right[1])
    return [notRob,rob]
}



function rob(root: TreeNode | null): number {
    const res: [number, number] = traversal(root)
    return Math.max(res[0],res[1])
};
```
