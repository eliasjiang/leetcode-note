let path:number[] = []
let result:number[][] = []
function dfs(graph:number[][],node:number):void{
    if(node === graph.length-1){
        result.push([...path])
        return 
    }

    for(let n of graph[node]){

        path.push(n)
        dfs(graph,n)
        path.pop()
    }
}
function allPathsSourceTarget(graph: number[][]): number[][] {
    path = [0]
    result = []
    dfs(graph,0)
    return result
};
