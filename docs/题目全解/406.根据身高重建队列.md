function reconstructQueue(people: number[][]): number[][] {
  people.sort((a, b) => {
    if (a[0] !== b[0]) {
        return b[0] - a[0]
    }else{
        return a[1] - b[1]
    }
  });

  let queue:number[][] = [];
  for(const p of people){
    let positon = p[1];
    queue.splice(positon,0,p);
  }
  return queue;

}
