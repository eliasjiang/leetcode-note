function numSquares(n: number): number {
  if (Math.sqrt(n) % 1 === 0) return 1;
  let len: number = Math.floor(Math.sqrt(n));
  let nums: number[] = new Array(len + 1);
  for (let i = 0; i <= len; i++) {
    nums[i] = i ** 2;
  }

  let dp: number[] = new Array(n + 1);
  dp[0] = 0;
  for (let i = 1; i <= n; i++) {
    dp[i] = i;
  }

  for (let i = 1; i <= len; i++) {
    for (let j = nums[i]; j <= n; j++) {
      dp[j] = Math.min(dp[j], dp[j - nums[i]] + 1);
    }
  }

  return dp[n];
}
