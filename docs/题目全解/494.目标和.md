
动态规划 滚动数组
```typescript
function findTargetSumWays(nums: number[], target: number): number {
    let sum:number = nums.reduce((pre, cur) => pre + cur, 0)
    if(Math.abs(target) > sum) return 0;
    if((target + sum) % 2 === 1) return 0;
    let bagSize:number = (target + sum) / 2;
    let dp:number[] = new Array(bagSize + 1).fill(0);
    dp[0] = 1;
    for(let i = 0; i < nums.length; i++) {

        for(let j = bagSize; j >= nums[i]; j--) {
            dp[j] += dp[j - nums[i]];
        }
    }
    return dp[bagSize];
};
```

动态规划 二维数组
```typescript
function findTargetSumWays(nums: number[], target: number): number {
    let sum:number = nums.reduce((pre, cur) => pre + cur, 0)
    if(Math.abs(target) > sum) return 0;
    if((target + sum) % 2 === 1) return 0;
    let t:number = (target + sum) / 2;
    let dp:number[][] = []
    for(let i = 0; i < nums.length + 1; i++) {
        dp.push(new Array(t + 1).fill(0))
    }
    
    if(nums[0]<=t){
        dp[0][nums[0]] = 1
    }

    let numZeros:number = 0

    for(let i = 0; i < nums.length; i++){
        if(nums[i] === 0){
            numZeros++;
        }
        dp[i][0] = Math.pow(2, numZeros);
    }

    for(let i = 1; i < nums.length; i++){
        for(let j=1;j<t+1;j++){
            if(j >= nums[i]){
                dp[i][j] = dp[i - 1][j] + dp[i - 1][j - nums[i]]
            }else{
                dp[i][j] = dp[i - 1][j]
            }
        }
    }

    return dp[nums.length-1][t]
};
```
