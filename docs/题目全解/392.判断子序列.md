
动态规划
本质就是最长子序列问题
```typescript
function isSubsequence(s: string, t: string): boolean {
    let dp:number[][] = new Array(s.length+1).fill(0).map(() => new Array(t.length+1).fill(0));

    for(let i=1;i<=s.length;i++){
        for(let j=1;j<=t.length;j++){
            if(s[i-1] === t[j-1]){
                dp[i][j] = dp[i-1][j-1] +1
            }else{
                dp[i][j] = dp[i][j-1]
            }
        }
    }

    return dp[s.length][t.length] === s.length
};
```

双指针

```typescript
function isSubsequence(s: string, t: string): boolean {
    let s_ptr: number = 0;
    let t_ptr: number = 0;
    let count: number = 0;
    while (t_ptr < t.length && s_ptr < s.length) {
        if (s[s_ptr] === t[t_ptr]) {
            s_ptr += 1;
            count += 1
        }
        t_ptr += 1;
    }
    return count === s.length;
}
```
