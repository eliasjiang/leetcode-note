
后序遍历 + 贪心策略
```typescript
enum state{
    COVERED,
    CAMERA,
    UNCOVERED
}
let result:number = 0;
function traversal(cur: TreeNode | null): state {
    
    if (cur === null) return state.COVERED;

    let left:number = traversal(cur.left)
    let right:number = traversal(cur.right)

    if (left == state.COVERED && right == state.COVERED) return state.UNCOVERED

    if (left == state.UNCOVERED || right == state.UNCOVERED){
        result ++;
        return state.CAMERA;
    }

    if(left == state.CAMERA || right == state.CAMERA){
        return state.COVERED;
    }


}
function minCameraCover(root: TreeNode | null): number {
    result = 0;
    if(traversal(root) == state.UNCOVERED) result ++;
    return result;
};
```
