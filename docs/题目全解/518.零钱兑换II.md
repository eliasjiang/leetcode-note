
动态规划 完全背包问题
```typescript
function change(amount: number, coins: number[]): number {
    let dp: number[] = new Array(amount + 1).fill(0)
    dp[0] = 1
    for (let coin of coins) {
        for (let i = coin; i <= amount; i++) {
            dp[i] += dp[i - coin]
        }
    }
    return dp[amount]
};
```
