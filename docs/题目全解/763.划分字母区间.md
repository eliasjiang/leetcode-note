function partitionLabels(s: string): number[] {
    let dictonary: { [key: string]: number } = {};
    for (let i = 0; i < s.length; i++) {
        dictonary[s[i]] = i;
    }
    let result:number[] = []
    let left:number = 0;
    let right:number = 0;

    for (let i = 0; i < s.length; i++) {
        right = Math.max(right, dictonary[s[i]]);
        if (i == right) {
            result.push(right - left + 1);
            left = right + 1;
        }
    }

    return result;
};

function cmp(a:number[],b:number[]):number{
    if (a[0]==b[0]){
        return a[1] - b[1]
    }
    return a[0] - b[0]
}

function countLabels(s:string):number[][]{
    let dictionary : {[key:string]:number[]} = {}
    for (let i = 0;i < s.length;i++){
        if(s[i] in dictionary){
            dictionary[s[i]].push(i)
        }else{
            dictionary[s[i]] = [i];
        }
    }
    let segment_list:number[][] = [];
    for(let key in dictionary){
        segment_list.push([dictionary[key][0],dictionary[key][dictionary[key].length - 1]]);
    }

    return segment_list.sort(cmp);
}
function partitionLabels(s: string): number[] {
    let res:number[] = [];
    let segment_list:number[][] = countLabels(s);
    let rightBoard:number = segment_list[0][1];
    let leftBoard:number = 0;

    for (let i = 0;i < segment_list.length;i++){
        if (segment_list[i][0] > rightBoard){
            res.push(rightBoard - leftBoard + 1);
            leftBoard = segment_list[i][0];
        }
            rightBoard = Math.max(rightBoard,segment_list[i][1]);
        
    }

    res.push(rightBoard - leftBoard + 1);
    return res;

};
